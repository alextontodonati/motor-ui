## Usage

import { Props } from 'docz'
import { Playground } from 'docz'
import Scatter from '../../lib/components/React/Scatter/Scatter.jsx';
import Motor from '../../lib/components/React/Motor/Motor.jsx';
import { config } from '../config/config.js'

<Playground>
<Motor  
 config={config}>  
       <Scatter
            cols={[
      { qField: 'country', qLabel: 'Country' },
      {
        qField: '=Count([variety])',
        qLabel: 'Varieties',
      },
      {
        qField: '=Count([winery])',
        qLabel: 'Wineries',
      },
    ]}
            height="410px"
            // border = {false}
            />
  </Motor>
</Playground>

## Theming


### Entry Point

```javascript
theme.scatter
```

### Object

Example below is taken from the Motor Theme.

```javascript

scatter: {
  /** Configuration object to connect to the Qlik Engine. Must include Qlik site URL and an App name */
  config: PropTypes.object,
  /** Scatter label  */
  label: PropTypes.string,
  /** cols from Qlik Data Model to render in the Scatter  */
  cols: PropTypes.array.isRequired,
  /** Calc condition for the chart  */
  calcCondition: PropTypes.object,
  /** Supress zeo vlaues in the the chart  */
  suppressZero: PropTypes.bool,
  /** Allow esizing of scrollbar for zoom */
  allowZoom: PropTypes.bool,
  /** Label Position */
  showLabels: PropTypes.oneOf(['top', 'none']),
  /** Allow text on the the chart axis */
  textOnChartAxis: PropTypes.bool,
  /** Supress scrollbar in the the chart  */
  suppressScroll: PropTypes.bool,
  /** Show Gridlines in Chart */
  showGridlines: PropTypes.bool,
  /** Scatter Sort Order */
  columnSortOrder: PropTypes.array,
  /** Sort Ascending or descending */
  sortDirection: PropTypes.string,
  /** Scatter theme try juno, classic or night. Default theme is 'classic' */
  theme: PropTypes.string,
  /** Scatter width */
  width: PropTypes.string,
  /** The height of the Scatter */
  height: PropTypes.string,
  /** The amount of margin around the component */
  margin: PropTypes.string,
  /** Whether the KPI size should scale for tablet and mobile */
  responsive: PropTypes.bool,
  /** Size of the Scatter */
  size: PropTypes.oneOf(['tiny', 'small', 'medium', 'large', 'xlarge']),
  /** Color of the Scatter Labels */
  fontColor: PropTypes.string,
  /** Border style of the Scatter chart */
  chartBorder: PropTypes.string,
  /** Background Color of the chart */
  backgroundColor: PropTypes.string,
  /** RoundNum of the Scatter */
  roundNum: PropTypes.bool,
  /** Title of the Scatter */
  title: PropTypes.string,
  /** Sub Title of the Scatter */
  subTitle: PropTypes.string,
  /** Legend of the Scatter */
  showLegend: PropTypes.bool,
  /** Offset of the selction panel */
  offset: PropTypes.number,
  },

```
