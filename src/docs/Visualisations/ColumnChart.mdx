import { Props } from 'docz'
import { Playground } from 'docz'
import Column from '../../lib/components/React/Column/Column.jsx'
import Motor from '../../lib/components/React/Motor/Motor.jsx'
import Filter from '../../lib/components/React/Filter/Filter.jsx'
import { config } from '../config/config.js'

<Playground>
<Motor  
 config={config}>  
  <Filter label='Country' dimension={['country']} />
          <Column
            cols={[
              { qField: 'country', qLabel: 'Country' },
              { qField: '=Avg(price)', qLabel: 'Average Price' },
             // { qField: '=sum(price)', qLabel: 'Total Price' }
            ]}
            chartColor={'divergent9'}
            size='small'
            showLabels='none'
            tickSpacing="wide"
            showGridlines='solid'
            responsive={true}
            columnPadding={10}
            stacked={false}
            height="410px"
            calcCondition={{
              qCond: '1+1=2',
              qMsg: 'Calculation condition has not been met'
            }}>
    </Column>
  </Motor>
</Playground>
